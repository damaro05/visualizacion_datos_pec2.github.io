# draw plot
pitch = VerticalPitch(half=True)
fig, ax = pitch.draw(figsize=(10, 10))

# Plot Voronoi
team1, team2 = pitch.voronoi(x, y, teams)
t1 = pitch.polygon(team1, ax=ax, fc='#c34c45', ec='black', lw=3, alpha=0.4)
t2 = pitch.polygon(team2, ax=ax, fc='#4584c3', ec='black', lw=3, alpha=0.4)

# Plot players
sc1 = pitch.scatter(x[teams], y[teams], ax=ax, c='#c34c45', s=150)
sc2 = pitch.scatter(x[~teams], y[~teams], ax=ax, c='#4584c3', s=150)

plt.show()
